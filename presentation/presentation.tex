\part[Code]{Displaying Code}
\section{Displaying Code}
\begin{frame}[fragile]{Displaying Code}
  \begin{minipage}{0.48\textwidth}
    \centering
    \begin{itemize}
    \item<1-> Code syntax highlighting provided by \texttt{minted} package
    \item<1-> You'll need to have \texttt{pygments} installed to highlight
    \item<2-> To highlight, use \texttt{minted} environment
    \item<2-> Use \mintinline{latex}{[fragile]} option in frame
    \item<2-> Don't put \texttt{minted} inside \texttt{invisible}, use code
      boxes (later)
    \item<3-> Display inline code with
      \mintinline{latex}{\mintinline{language}{some code}}:
      \mintinline{python}{l = [x for x in range(0, 9)]} (python)
    \end{itemize}
  \end{minipage}
  \hfill
  \begin{minipage}{0.48\textwidth}
    \colorheading{Python:}
    \begin{minted}[autogobble]{python}
      def hello():
        print('Hello World!')
    \end{minted}
    \colorheading{C++:}
    \begin{minted}[autogobble]{cpp}
      using namespace std;
      int main(void) {
        cout << "Hello World!" << endl;
        return 0;
      }
    \end{minted}
  \end{minipage}
\end{frame}

\begin{frame}[fragile]{Code in boxes}
  \begin{minipage}{0.45\textwidth}
    \centering
    \begin{itemize}
      \item<1-> Display code inside boxes uses \texttt{tcolorbox} and
        \texttt{minted} together
      \item<1-> Beware that using \texttt{codeboxtc} still will break with \texttt{
          invisible, only
        }
      \item<2-> For finer control, use \texttt{BeamerShowCode}
      \item<3-> Highlight code lines
    \end{itemize}
  \end{minipage}
  \begin{minipage}{0.5\textwidth}
    \begin{center}
    \begin{codeboxtc}{python}{Example Python Code}{}{}
      class MyClass:
        def __init__(self):
          raise RuntimeError('I am useless')
    \end{codeboxtc}
    \invisible<1>{
      \begin{overprint}
        \onslide<2>\BeamerShowCode{
          minipage width=1.0\textwidth, box scale=1.0,
          listing file={file.cpp}, language=cpp}{C++ code loaded from file}{}
        \onslide<3>\BeamerShowCode{
          minipage width=1.0\textwidth, box scale=1.0,
          listing file={file.cpp}, language=cpp}{C++ code loaded from file}{3-4}
      \end{overprint}}
   \end{center}
  \end{minipage}
\end{frame}
